# ── Baseline 2: Flat FedAvg (No Hierarchy) ───────────────────────────────────
# Standard FedAvg — same clients, same rounds, no cluster structure, no attack.
# Shows what the hierarchical structure contributes beyond vanilla federation.

if ray.is_initialized():
    ray.shutdown()

_flat_seed = 42
tf.keras.backend.clear_session()
_flat_init_model = BeamModel(nr, nt, beam_codebook_size)
_flat_init_model.build_model(nr, nt)


def flat_client_fn(context: fl.common.Context) -> fl.client.Client:
    _cidx = int(context.node_id) % len(clients)
    _cobj = clients[_cidx]
    tf.config.threading.set_intra_op_parallelism_threads(1)
    tf.config.threading.set_inter_op_parallelism_threads(1)
    tf.random.set_seed(_flat_seed + _cidx)
    np.random.seed(_flat_seed + _cidx)
    _model = BeamModel(nr, nt, beam_codebook_size)
    _model.build_model(nr, nt)
    _client = BeamFlowerClient(
        model=_model, dataset=ds,
        train_indices=_cobj.train_indices, test_indices=_cobj.test_indices,
        cfg=CFG, trajectory_id=_cobj.trajectory_id,
        beam_changes=beam_changes_global,
    )
    _client.tx_codebook = ds.tx_codebook
    _client.rx_codebook = ds.rx_codebook
    _client.cache_dir   = PRECOMPUTE_CACHE_DIR
    _client.client_idx  = _cidx
    return _client.to_client()


flat_strategy = fl.server.strategy.FedAvg(
    fraction_fit          = CFG["client_frac"],
    fraction_evaluate     = CFG["client_frac"],
    min_fit_clients       = len(clients),
    min_evaluate_clients  = len(clients),
    min_available_clients = len(clients),
    initial_parameters    = fl.common.ndarrays_to_parameters(_flat_init_model.get_weights()),
    fit_metrics_aggregation_fn      = agg_metrics,
    evaluate_metrics_aggregation_fn = agg_metrics,
)

FLAT_ROUNDS = 30
print(f"\nFlat FedAvg Baseline: {FLAT_ROUNDS} rounds, {len(clients)} clients (no attack)")

flat_history = fl.simulation.start_simulation(
    client_fn        = flat_client_fn,
    num_clients      = len(clients),
    config           = fl.server.ServerConfig(num_rounds=FLAT_ROUNDS),
    strategy         = flat_strategy,
    client_resources = {"num_cpus": 1, "num_gpus": 0.0},
)
print("Flat FedAvg complete.")

FLAT_ACC  = float("nan")
FLAT_TOP3 = float("nan")
if flat_history.metrics_distributed:
    _vals = flat_history.metrics_distributed.get("beam_accuracy", [])
    if _vals: FLAT_ACC = max(v for _, v in _vals)
    _vals = flat_history.metrics_distributed.get("beam_top3_accuracy", [])
    if _vals: FLAT_TOP3 = max(v for _, v in _vals)
    print(f"  Best beam_accuracy      : {FLAT_ACC:.4f}")
    print(f"  Best beam_top3_accuracy : {FLAT_TOP3:.4f}")
